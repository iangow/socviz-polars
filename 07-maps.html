7  Draw Maps

This chapter follows Chapter 7 of Healy (2026), translating the main map examples to Python, Polars, and plotnine_polars. The Python setup here is intentionally lightweight: we use the preprojected polygon tables from socviz and draw them with ordinary plotnine polygon layers. That keeps the data work in Polars and avoids adding a full GIS stack for this introductory chapter. The code used to prepare the map-related Parquet files lives in the data-raw/ directory of the socviz_py package.

import polars as pl

from plotnine_polars import aes
import plotnine_polars as p9
from socviz_pl import load_data

def theme_map():
    return (
        p9.theme_void()
        + p9.theme(
            dpi=100,
            legend_position="bottom",
            plot_title=p9.element_text(ha="left", size=12),
            strip_background=p9.element_rect(fill="white", color="white"),
        )
    )
election24 = load_data("election24").with_columns(
    fips_str=pl.col("fips").cast(pl.Utf8).str.zfill(2),
    region=pl.col("state").str.to_lowercase(),
)

county_map = load_data("county_map").with_columns(
    state_fips=pl.col("id").str.slice(0, 2),
)
us_states = load_data("us_states")
county_data = load_data("county_data")

opiates = (
    load_data("opiates")
    .with_columns(
        year=pl.col("year").cast(pl.Int64),
        state_fips=pl.col("fips"),
    )
    .join(
        election24.select("fips_str", "state"),
        left_on="state_fips",
        right_on="fips_str",
        how="left",
    )
    .with_columns(region=pl.col("state").str.to_lowercase())
)

state_vertices = con.sql("""
WITH 
raw_parts AS (
  SELECT
    state,
    dump.geom AS geom,
    dump.path AS path
  FROM states,
  UNNEST(ST_Dump(geom)) AS t(dump)
),
parts AS (
  SELECT
    state,
    geom,
    row_number() OVER (
      PARTITION BY state
      ORDER BY CAST(path AS VARCHAR)
    ) AS part_id
  FROM raw_parts
),
rings AS (
  SELECT
    state,
    state || '_' || part_id::VARCHAR AS group_id,
    ST_ExteriorRing(geom) AS ring
  FROM parts
),
points AS (
  SELECT
    state,
    group_id,
    i AS ord,
    ST_PointN(ring, i::INTEGER) AS pt
  FROM rings,
  generate_series(1, ST_NumPoints(ring)::INTEGER) AS t(i)
)
SELECT
  state,
  group_id,
  ord,
  ST_X(pt) AS x,
  ST_Y(pt) AS y
FROM points
ORDER BY state, group_id, ord
""").pl()

Once the data are joined, we can fill the polygons by the winning party. The important practical point is the same as in ggplot: make sure the keys used for the join really match. Here those keys are the lower-case state names in region.

(
    elections24_sf
    .ggplot(aes(x="x", y="y", group="group_id", fill="party"))
    .geom_polygon(color="white", size=0.05)
    .coord_equal()
    .scale_fill_manual(values=party_colors)
    .labs(fill=None)
)

With the map data frame in place, we can map continuous variables too. The default continuous fill is not ideal for election data, so we specify the gradient directly.

(
    elections24_sf
    .ggplot(aes(x="x", y="y", group="group_id", fill="pct_trump"))
    .geom_polygon(color="white", size=0.05)
    .coord_equal()
    .scale_fill_gradient(low="white", high="#CB454A", labels=lambda lst: [f"{v:.0%}" for v in lst])
    .labs(fill="Percent")
)

For margins, a diverging scale is more appropriate. The d_points variable is positive when Harris’s margin is larger and negative when Trump’s is larger.

(
    elections24_sf
    .filter(pl.col("st") != "DC")
    .ggplot(aes(x="x", y="y", group="group_id", fill="d_points"))
    .geom_polygon(color="white", size=0.05)
    .coord_equal()
    .scale_fill_gradient2(low="red", mid="#756BB1", high="blue")
    .labs(fill="Percent")
)

7.6 Is Your Data Really Spatial?

Even if data are collected in spatial units, a map may not be the clearest display. The same opiate data can be shown as time series, grouped by Census division. This makes the state trajectories and regional variation easier to compare.

p9.theme_set(p9.theme_gray())
<plotnine.themes.theme_void.theme_void at 0x126c2adf0>
(
    opiates
    .filter(pl.col("st") != "DC")
    .ggplot(aes(x="year", y="adjusted", group="st"))
    .geom_line(color="#B3B3B3")
    .labs(x="Year", y="Rate per 100,000 population")
)
division_means = (
    opiates
    .filter(pl.col("division_name").is_not_null())
    .group_by("division_name")
    .agg(pl.col("adjusted").mean().alias("division_mean"))
)

division_order = (
    division_means
    .sort("division_mean", descending=True)
    .get_column("division_name")
    .to_list()
)

opiates_div = (
    opiates
    .filter(pl.col("division_name").is_not_null())
    .join(division_means, on="division_name", how="left")
    .with_columns(division_ordered=pl.col("division_name").cast(pl.Enum(division_order)))
)

opiates_labels = opiates_div.filter(pl.col("year") == pl.col("year").max().over("st"))
(
    opiates_div
    .ggplot(aes(x="year", y="adjusted"))
    .geom_line(aes(group="st"), color="#B3B3B3")
    .geom_smooth(aes(group="division_name"), se=False)
    .geom_text(
        data=opiates_labels,
        mapping=aes(x="year", y="adjusted", label="st"),
        size=6,
        nudge_x=0.45,
    )
    .facet_wrap("division_ordered", nrow=3)
    .labs(
        x="",
        y="Rate per 100,000 population",
        title="State-Level Opiate Death Rates by Census Division, 1999-2019",
    )
    .theme_minimal()
    .add_theme(strip_background=p9.element_blank())
)

7.7 Where to Go Next

These examples use polygon tables as ordinary data frames, which is enough to reproduce the main lessons of the chapter. For more serious map work in Python, the next step would be a spatial data stack such as GeoPandas, Shapely, and PyProj. The conceptual warning remains the same: maps are compelling, but they are not always the clearest or most honest way to show spatially grouped social data.
